#define DHTPIN 27      
#define DHTTYPE DHT11
#define MQ_PIN 32      
#define LDR_PIN 34     
#define PIR_PIN 13     
#define LED_PIN 26          // LED đỏ báo động Gas
#define BUZZER_PIN 33       // Còi báo động
#define RELAY_FAN_PIN 14    // Relay 1: Quạt (Hút Gas / Làm mát)
#define RELAY_LIGHT_PIN 25  // Relay 2: Đèn phòng tự động

DHT dht(DHTPIN, DHTTYPE);

// ---QUAT & DEN ---
String webCommandQuat="OFF";
String webCommandDen="OFF";
// --- BIẾN ĐO LƯỜNG & LOGIC ---
unsigned long tgFirebase = 0, tgDHT = 0, tgDashboard = 0, tgLichSu = 0;
unsigned long thoiGianKhoiDong, thoiGianChuyenDongCuoi = 0;
float t = 0, h = 0; 
int gas = 0, ldr = 0, anhSangPct = 0;
bool dangCoNguoi = false, dangBaoDongCoi = false;
String lastGasStat = "", lastPresStat = ""; 

// --- CÁC HÀM PHÂN LOẠI TRẠNG THÁI ---
String getTempStatus(float val) {
  if (val < 22) return "Se lanh";
  if (val <= 32) return "Ly tuong";
  if (val <= 37) return "Nong";
  return "RAT NONG";
}

String getHumidStatus(float val) {
  if (val < 40) return "Kho hanh";
  if (val <= 70) return "Thoai mai";
  return "Am uot";
}

String getLightStatus(int pct) {
  if (pct < 30) return "Toi";
  if (pct <= 70) return "Anh sang vua";
  return "Rat sang";
}

String getGasStatus(int val) {
  if (val < 400) return "An toan";
  if (val <= 600) return "Canh bao";
  return "🚨 NGUY HIEM";
}

// --- CÁC HÀM TRỢ GIÚP THỜI GIAN ---
String layGio() {
  struct tm ti;
  if(!getLocalTime(&ti)) return "Cho dong bo...";
  char buf[25];
  strftime(buf, sizeof(buf), "%H:%M:%S %d/%m/%Y", &ti);
  return String(buf);
}

String layNgay() {
  struct tm ti;
  if(!getLocalTime(&ti)) return "2026-03-20";
  char buf[15];
  strftime(buf, sizeof(buf), "%Y-%m-%d", &ti);
  return String(buf);
}

int docAnalog(int p) {
  long s = 0;
  for (int i = 0; i < 20; i++) { s += analogRead(p); delay(1); }
  return s / 20;
}

// ==========================================
// 2. THIẾT LẬP (SETUP)
// ==========================================
void setup() {
  WRITE_PERI_REG(RTC_CNTL_BROWN_OUT_REG, 0); 
  Serial.begin(115200);
  dht.begin();
  
  pinMode(PIR_PIN, INPUT); 
  pinMode(LED_PIN, OUTPUT);
  pinMode(BUZZER_PIN, OUTPUT);
pinMode(RELAY_FAN_PIN, OUTPUT);
  pinMode(RELAY_LIGHT_PIN, OUTPUT);

  digitalWrite(LED_PIN, LOW); 
  digitalWrite(BUZZER_PIN, HIGH); 
  digitalWrite(RELAY_FAN_PIN, HIGH);  
  digitalWrite(RELAY_LIGHT_PIN, HIGH);

  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  Serial.print("\n[WIFI] Ket noi");
  while (WiFi.status() != WL_CONNECTED) { delay(500); Serial.print("."); }
  
  configTime(7 * 3600, 0, "pool.ntp.org");
  config.api_key = API_KEY;
  config.database_url = DATABASE_URL;
  auth.user.email = USER_EMAIL;
  auth.user.password = USER_PASSWORD;
  Firebase.begin(&config, &auth);

  thoiGianKhoiDong = millis(); 
  Serial.println("\n✅ HE THONG V32.0 SAN SANG (BOOTING 30S...)");
}

// ==========================================
// 3. VÒNG LẶP CHÍNH (LOOP)
// ==========================================
void loop() {
  
  gas = docAnalog(MQ_PIN);
  ldr = docAnalog(LDR_PIN);
  anhSangPct = map(ldr, 0, 4095, 100, 0);
  
  unsigned long thoiGianTroiQua = millis() - thoiGianKhoiDong;
  bool daWarmUpXong = (thoiGianTroiQua > 30000);

  if (digitalRead(PIR_PIN)) { thoiGianChuyenDongCuoi = millis(); dangCoNguoi = true; }
  if (dangCoNguoi && (millis() - thoiGianChuyenDongCuoi > 300000)) dangCoNguoi = false;

  if (millis() - tgDHT > 2000) {
    tgDHT = millis();
    float tn = dht.readTemperature();
    float hn = dht.readHumidity();
    if(!isnan(tn) && !isnan(hn)) { t = tn; h = hn; }
  }

  String curGasStat = getGasStatus(gas);
  String curPresStat = dangCoNguoi ? "CO NGUOI" : "Trong";
  String thoiGianBayGio = layGio();

  // ==========================================
  // D1. MODULE CẢNH BÁO & LÀM MÁT (SAFETY & CLIMATE)
  // ==========================================
  if (daWarmUpXong) {
    
    // --- 1. Logic Còi và Đèn (Chỉ phụ thuộc vào Gas) ---
    if (gas > 600) dangBaoDongCoi = true; 
    else if (gas < 500) dangBaoDongCoi = false;

    if (dangBaoDongCoi) {
      digitalWrite(BUZZER_PIN, LOW);   // HÚ CÒI
      digitalWrite(LED_PIN, HIGH);     // SÁNG ĐÈN ĐỎ
    } else if (gas > 400) {
      digitalWrite(BUZZER_PIN, HIGH);  // TẮT CÒI
      digitalWrite(LED_PIN, HIGH);     // VẪN SÁNG ĐÈN ĐỎ CẢNH BÁO SỚM
    } else {
      digitalWrite(BUZZER_PIN, HIGH);
      digitalWrite(LED_PIN, LOW);
    }

    // --- 2. Logic Quạt Thông Minh (Gas HOẶC [Nhiệt Độ + Hiện Diện]) ---
    static bool quatDangChay = false;
    const float NGUONG_NONG = 33.5;

    // BƯỚC 1: Xác định khi nào CẦN BẬT
    // Bật nếu: Gas cao HOẶC (Nóng + Có người) HOẶC (Web nhấn ON)
    if (gas > 400 || (t >= NGUONG_NONG && dangCoNguoi) || (webCommandQuat == "ON")) {
      quatDangChay = true; 
    } 
    // BƯỚC 2: Xác định khi nào BUỘC TẮT (Cưỡng ép từ Web)
    else if (webCommandQuat == "OFF") {
      quatDangChay = false;
    }
    // BƯỚC 3: Nếu không có lý do gì để bật (Gas giảm, trời mát) -> TẮT
    else {
      quatDangChay = false; 
    }

    if (quatDangChay) {
      digitalWrite(RELAY_FAN_PIN, LOW); // Bật Relay Quạt
    } else {
      digitalWrite(RELAY_FAN_PIN, HIGH);// Tắt Relay Quạt
    }
  }

  // ==========================================
  // D2. MODULE TỰ ĐỘNG HÓA (LIGHTING)
  // ==========================================
  if (daWarmUpXong && !dangBaoDongCoi) { 
    if ((ldr > 3000 && dangCoNguoi) || (webCommandDen == "ON")) {
      digitalWrite(RELAY_LIGHT_PIN, LOW); // Bật đèn phòng
    } else {
      digitalWrite(RELAY_LIGHT_PIN, HIGH); // Tắt đèn phòng
    }
  }

  // ==========================================
  // E. ĐỒNG BỘ DỮ LIỆU FIREBASE
  // ==========================================
  if (daWarmUpXong && Firebase.ready()) {
    if (Firebase.RTDB.getString(&fbdo, "SmartHome/HienTai/ThietBi/Quat")) {
    webCommandQuat = fbdo.stringData(); // webCommandQuat là biến String
    } 
    if (Firebase.RTDB.getString(&fbdo, "SmartHome/HienTai/ThietBi/Den")) {
    webCommandDen = fbdo.stringData();
    } 
    // 1. Cập nhật Hiện Tại (5s/lần)
    if (millis() - tgFirebase > 5000) {
      tgFirebase = millis();
      FirebaseJson cur;
      cur.set("MoiTruong/NhietDo/Val", t); cur.set("MoiTruong/NhietDo/Status", getTempStatus(t));
      cur.set("MoiTruong/DoAm/Val", h); cur.set("MoiTruong/DoAm/Status", getHumidStatus(h));
      cur.set("MoiTruong/AnhSang/Pct", anhSangPct); cur.set("MoiTruong/AnhSang/Status", getLightStatus(anhSangPct));
      cur.set("AnToan/Gas/Val", gas); cur.set("AnToan/Gas/Status", curGasStat);
      cur.set("HienDien", curPresStat);
      cur.set("ThietBi/Quat", (digitalRead(RELAY_FAN_PIN) == LOW) ? "ON" : "OFF");
      cur.set("ThietBi/Den", (digitalRead(RELAY_LIGHT_PIN) == LOW) ? "ON" : "OFF");
      cur.set("SyncTime", thoiGianBayGio);
      
      Firebase.RTDB.setJSON(&fbdo, "SmartHome/HienTai", &cur);
    }

    // 2. Nhật ký sự kiện 
    if (curGasStat != lastGasStat || curPresStat != lastPresStat) {
      FirebaseJson log;
      log.set("Time", thoiGianBayGio);
      log.set("Event", (curGasStat == "🚨 NGUY HIEM") ? "CANH BAO GAS" : "THAY DOI TRANG THAI");
      log.set("Detail", "Gas: " + curGasStat + " | PIR: " + curPresStat);
      Firebase.RTDB.pushJSON(&fbdo, "SmartHome/NhatKy_KhanCap", &log);
      lastGasStat = curGasStat; lastPresStat = curPresStat;
    }

    // 3. Lịch sử định kỳ (1 phút/lần)
    if (millis() - tgLichSu > 60000) {
      tgLichSu = millis();
      FirebaseJson hist;
      hist.set("Time", thoiGianBayGio); hist.set("Temp", t); 
      hist.set("Humid", h); hist.set("Gas", gas); hist.set("Light", anhSangPct);
      Firebase.RTDB.pushJSON(&fbdo, "SmartHome/LichSu/" + layNgay(), &hist);
    }
  }

  // ==========================================
  // F. DASHBOARD (SERIAL MONITOR)
  // ==========================================
  if (millis() - tgDashboard > 2000) {
    tgDashboard = millis();
    Serial.println("\n================ DASHBOARD V32.0 ================");
if (daWarmUpXong) {
      Serial.printf("⏰ %s | 🔥 Gas: %d (%s)\n", thoiGianBayGio.c_str(), gas, curGasStat.c_str());
      Serial.printf("🌡 %0.1fC (%s) | 💧 %0.1f%% (%s)\n", t, getTempStatus(t).c_str(), h, getHumidStatus(h).c_str());
      Serial.printf("💡 Sáng: %d%% (%s) | 👥: %s\n", anhSangPct, getLightStatus(anhSangPct).c_str(), curPresStat.c_str());
      
      // Chú thích lý do quạt chạy chi tiết hơn
      String lyDoQuat = "OFF";
      if (digitalRead(RELAY_FAN_PIN) == LOW) {
        if (gas > 400 && t >= 33.5 && dangCoNguoi) lyDoQuat = "ON (Hút Gas & Làm Mát)";
        else if (gas > 400) lyDoQuat = "ON (Hút Gas uu tien!)";
        else lyDoQuat = "ON (Làm Mát do có người)";
      }
      Serial.printf("⚙️ Quat: %s | ⚙️ Den: %s\n", lyDoQuat.c_str(), (digitalRead(RELAY_LIGHT_PIN) == LOW)?"ON":"OFF");
    } else {
      Serial.printf("⏳ KHOI DONG: Con %lu giay...\n", (30000 - thoiGianTroiQua) / 1000);
    }
    Serial.println("=================================================");
  }
}
